  GpuDispatcher::run():
  while (running_) {
      // Round-Robin 轮询 3 路 RawCPI_Q
      for (uint8_t i = 0; i < GPU_STREAM_COUNT; ++i) {
          uint8_t q_idx = (rr_index_ + i) % GPU_STREAM_COUNT;

          // 1. 先 poll 该 stream 的完成结果
          PlotBatch *result = nullptr;
          if (pipeline_.poll_completion(q_idx, &result) && result) {
              plots_qs_[q_idx].drop_oldest_push(result);
              metrics_.plots_produced++;
          }

          // 2. 如果 inflight 未满，尝试从队列取 RawBlock 并 submit
          if (pipeline_.inflight_count(q_idx) < GPU_INFLIGHT_MAX) {
              auto opt = rawcpi_qs_[q_idx].try_pop();
              if (opt) {
                  auto ec = pipeline_.submit_cpi(*opt, q_idx);
                  if (ec != ErrorCode::OK) {
                      metrics_.submit_errors++;
                  }
              }
          }
      }
      rr_index_ = (rr_index_ + 1) % GPU_STREAM_COUNT;

      // 3. 超时检查
      // ...（对每个 stream 检查 inflight 时长 > T_GPU_MAX_MS）
  }

═══ 交付物 2：5-State FSM ═══

文件：src/control/orchestrator/state_machine.cpp

  class StateMachine {
  public:
      enum class State { INIT, STANDBY, RUNNING, DEGRADED, FAULT };

      // 合法转换规则（严格，其余全部拒绝）：
      // INIT → STANDBY:    BootChecker 全部通过
      // STANDBY → RUNNING: 收到 START 命令
      // RUNNING → STANDBY: 收到 STOP 命令
      // RUNNING → DEGRADED: 单模块故障超时 T1_DEGRADED_MS=500ms 未恢复
      // DEGRADED → RUNNING: 故障模块恢复
      // DEGRADED → FAULT:  多模块故障或单模块超时 T2_FAULT_MS=2000ms
      // FAULT → INIT:      收到 RESET 命令

      bool transition(State target, const std::string& reason);
      State current() const;

      // 注册状态变化回调（通知 M04 推送 StateSnapshot）
      using Callback = std::function<void(State from, State to, const std::string& reason)>;
      void on_transition(Callback cb);
  };

═══ 交付物 3：Orchestrator — 模块生命周期管理 ═══

文件：src/control/orchestrator/orchestrator.cpp

  class Orchestrator {
  public:
      Orchestrator(StateMachine& fsm, EventDispatcher& events);

      // 启动序列：BootChecker → 逐模块 init → STANDBY
      void boot();
      // 运行序列：START → M01.start() → M02.start() → M03.start() → M04.start() → RUNNING
      void start();
      // 停止序列：STOP → M04.stop() → M03.stop() → M02.stop() → M01.stop() → STANDBY
      void stop();
      // 重置：RESET → 全模块 stop → 重新 boot
      void reset();

      // 注册模块（M01-M04 的 start/stop/health 接口）
      struct ModuleHandle {
          std::string name;
          std::function<bool()> start;
          std::function<void()> stop;
          std::function<bool()> is_healthy;
      };
      void register_module(ModuleHandle handle);
  };

      enum class EventType { COMMAND, STATE_CHANGE, MODULE_HEALTH, METRIC_ALERT, CONFIG_UPDATE };
      using Handler = std::function<void(const Event&)>;
      void subscribe(EventType type, Handler handler);
      void publish(Event event);
  };
