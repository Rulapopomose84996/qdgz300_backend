═══ 交付物 2：实现文件 ═══

文件：src/m01_receiver/delivery/rawblock_adapter.cpp

字段映射规则（必须严格遵守）：
  block->ingest_ts   = std::chrono::steady_clock::now() 转换为 ns（epoch_ns）
  block->data_ts     = packet.packet.header.timestamp（已是 ms，需转 ns：* 1000000）
  block->array_id    = 构造时传入的 array_id_（不从 header 取，因为 adapter 按阵面实例化）
  block->cpi_seq     = packet.sequence_number
  block->fragment_count = packet.packet.data_header.total_frags
                          （注意：OrderedPacket 中 ParsedPacket 是否包含 data_header 需确认，
                           若 OrderedPacket 是 Reorderer 输出的合成单包，total_frags 可能不可用——
                           此时使用 1 或从 header 推导）
  block->data_size   = static_cast<uint32_t>(packet.payload_size)
  block->flags       = 0
                       | (packet.is_zero_filled ? RawBlockFlags::INCOMPLETE_FRAME : 0)

  payload 拷贝：
  size_t copy_size = std::min(packet.payload_size, static_cast<size_t>(RAW_BLOCK_PAYLOAD_SIZE));
  if (packet.owned_payload && copy_size > 0) {
      std::memcpy(block->payload, packet.owned_payload.get(), copy_size);
  }
  if (packet.payload_size > RAW_BLOCK_PAYLOAD_SIZE) {
      block->flags |= static_cast<uint32_t>(RawBlockFlags::INCOMPLETE_FRAME);
      // 截断警告日志
  }

  推送策略：
  使用 queue_->drop_oldest_push(block) — 即使队列满也能推入（覆盖最旧数据）。

实现要点：
  - 使用 std::make_shared<RawBlock>() 分配（RawBlock 含 2MB payload 数组，每次分配约 2.1MB）
  - 不引入 mutex — SPSC 队列本身无锁

   - DeliveryInterface::deliver() 接受 const OrderedPacket&，不消耗 owned_payload

   - 因此先 deliver() 再 move() 是安全的——deliver() 只读 payload，不 move 走
3. **空检查**：rawblock_adapter 为 nullptr 时（consumer 未启用）直接跳过，零开销
4. **无锁保证**：路径 B 不需要 delivery_mutex——每阵面 SPSC 队列独立，RawBlockAdapter 不共享

═══ 硬约束 ═══
- DeliveryInterface::deliver(const OrderedPacket&) 在 RawBlockAdapter::adapt_and_push(&&) 之前调用
- deliver() 不可依赖 owned_payload 的所有权（它接受 const ref，仅读取）
- adapt_and_push() 内部做 memcpy payload，之后 owned_payload 是否被 move 走不影响
- 如果 delivery_interface 为 nullptr 且 rawblock_adapter 也为 nullptr，整个 lambda 是 no-op
- 不得在 lambda 内引入新的 mutex

  ErrorCode GpuPipeline::submit_cpi(RawBlock *block, uint8_t stream_id) noexcept
  {
      if (!block) return ErrorCode::GPU_H2D_FAILED;
      if (stream_id >= GPU_STREAM_COUNT) return ErrorCode::GPU_H2D_FAILED;

      // 查找空闲 slot
      int8_t slot_idx = find_free_slot(stream_id);
      if (slot_idx < 0) return ErrorCode::GPU_H2D_FAILED; // inflight 已满

      auto &slot = inflight_[stream_id][slot_idx];

  #ifndef QDGZ300_HAS_GPU
      // ═══ CPU Fallback 路径 ═══

      // 1. 将 payload 视为交替 float32 I/Q 样本
      const auto *iq = reinterpret_cast<const float *>(block->payload);
      const size_t num_samples = block->data_size / (2 * sizeof(float)); // I+Q pairs

      // 2. 简单峰值检测（幅度门限 CFAR-lite）
      constexpr float AMPLITUDE_THRESHOLD = 100.0f;
      constexpr size_t MAX_PLOTS_PER_CPI = 512;
      // 临时存放检测到的峰值
      std::vector<Plot> detected;           // 栈上 vector — 仅 CPU fallback 可用
      detected.reserve(std::min(num_samples, MAX_PLOTS_PER_CPI));

      for (size_t i = 0; i < num_samples && detected.size() < MAX_PLOTS_PER_CPI; ++i) {
          float I = iq[2 * i];
          float Q = iq[2 * i + 1];
          float amplitude = std::sqrt(I * I + Q * Q);
          if (amplitude > AMPLITUDE_THRESHOLD) {
              Plot p{};
              p.range_m = static_cast<double>(i);        // 简化：以采样点索引代替真实距离
              p.azimuth_deg = 0.0;
              p.elevation_deg = 0.0;
              p.doppler_mps = 0.0;
              p.snr_db = 20.0f * std::log10f(amplitude / AMPLITUDE_THRESHOLD);
              p.amplitude = amplitude;
              p.beam_id = 0;
              p.array_id = block->array_id;
              detected.push_back(p);
          }
      }

      // 3. 构造 PlotBatch
      auto *batch = new PlotBatch{};  // 生命周期由消费者释放
      batch->data_ts = block->data_ts;
      batch->process_ts = static_cast<uint64_t>(
          std::chrono::steady_clock::now().time_since_epoch().count());
      batch->array_id = block->array_id;
      batch->cpi_seq = block->cpi_seq;
      batch->plot_count = static_cast<uint16_t>(detected.size());
      batch->flags = (block->flags & RawBlock::FLAG_INCOMPLETE_FRAME)
                     ? RawBlock::FLAG_INCOMPLETE_FRAME : 0;

      // 4. 分配 Plot 数组（aligned_alloc 或 new[]）
      if (!detected.empty()) {
          batch->plots = static_cast<Plot *>(
              aligned_alloc(64, detected.size() * sizeof(Plot)));
          std::memcpy(batch->plots, detected.data(), detected.size() * sizeof(Plot));
      } else {
          batch->plots = nullptr;
      }

      // 5. 填充 inflight slot（CPU fallback 同步完成，直接标记 completed）
      slot.input = block;
      slot.output = batch;
      slot.occupied = true;
      slot.completed = true;  // CPU 模式下同步完成
      slot.submit_time = std::chrono::steady_clock::now();

      return ErrorCode::OK;

  #else
      // GPU 路径（后续实现）
      (void)block; (void)stream_id;
      return ErrorCode::OK;
  #endif
  }

  void GpuPipeline::handle_timeout(uint8_t stream_id) noexcept
  {
      if (stream_id >= GPU_STREAM_COUNT) return;

      auto now = std::chrono::steady_clock::now();

      for (uint8_t s = 0; s < GPU_INFLIGHT_MAX; ++s) {
          auto &slot = inflight_[stream_id][s];
          if (!slot.occupied) continue;

          auto elapsed = std::chrono::duration_cast<std::chrono::milliseconds>(
              now - slot.submit_time).count();

          if (elapsed > T_GPU_MAX_MS) {
              spdlog::warn("[M02] GPU timeout on stream {} slot {} ({}ms > {}ms)",
                           stream_id, s, elapsed, T_GPU_MAX_MS);

              // 如果有部分输出，设 FLAG_GPU_TIMEOUT
              if (slot.output) {
                  slot.output->flags |= RawBlock::FLAG_GPU_TIMEOUT;
                  // 不释放——让消费者处理（带 FLAG 标记的 batch）
                  // 但如果要丢弃：free_plot_batch(slot.output);
              }

              // 清空 slot
              slot.input = nullptr;
              slot.output = nullptr;
              slot.occupied = false;
              slot.completed = false;
          }
      }
  }

              auto elapsed = std::chrono::duration_cast<std::chrono::milliseconds>(

              if (elapsed > T_GPU_MAX_MS) return true;
          }
      }
      return false;
  }

  void GpuDispatcher::run() noexcept
  {
      // 可选：绑核到 CPU 19，设实时优先级
      // bind_thread_to_cpu(19);

      while (running_.load(std::memory_order_relaxed))
      {
          bool did_work = false;

          for (uint8_t i = 0; i < GPU_STREAM_COUNT; ++i)
          {
              uint8_t q_idx = (rr_index_ + i) % GPU_STREAM_COUNT;

              // ── Step 1: Poll 该 stream 的完成结果 ──
              PlotBatch *result = nullptr;
              if (pipeline_.poll_completion(q_idx, &result))
              {
                  if (result)
                  {
                      plots_qs_[q_idx].drop_oldest_push(result);
                      metrics_.gpu_frames_completed.increment();
                      did_work = true;
                  }
              }

              // ── Step 2: 超时检测 ──
              if (pipeline_.has_timeout(q_idx))
              {
                  pipeline_.handle_timeout(q_idx);
                  metrics_.gpu_timeout_total.increment();
              }

              // ── Step 3: 如果 inflight 未满，从队列取 RawBlock 并 submit ──
              if (pipeline_.inflight_count(q_idx) < GPU_INFLIGHT_MAX)
              {
                  auto opt = rawcpi_qs_[q_idx].try_pop();
                  if (opt.has_value())
                  {
                      RawBlock *block = opt.value();
                      auto ec = pipeline_.submit_cpi(block, q_idx);
                      if (ec == ErrorCode::OK)
                      {
                          metrics_.gpu_frames_submitted.increment();
                          did_work = true;
                      }
                      else
                      {
                          spdlog::warn("[M02] submit_cpi failed on stream {}: {}",
                                       q_idx, error_name(ec));
                          // block 的所有权需归还或释放——取决于设计
                          // 本实现中 block 指针来自 M01 的 SPSC 队列
                          // 如果 submit 失败，block 不再被使用
                      }
                  }
              }
          }

          // 递进 Round-Robin 索引
          rr_index_ = (rr_index_ + 1) % GPU_STREAM_COUNT;

          // ── Step 4: 如果本轮无工作，短暂 yield 避免空转 ──
          if (!did_work)
          {
              std::this_thread::sleep_for(std::chrono::microseconds(100));
          }
      }
  }
